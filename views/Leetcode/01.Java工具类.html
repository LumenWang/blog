    以上String、Integer等类已经实现了PriorityQueue的Comparable接口，但自定义类没有。所以自定义类需要自己实现 如

    import java.util.Comparator;
import java.util.PriorityQueue;
import java.util.Queue;

Queue<User> q = new PriorityQueue<>(new UserComparator());
q.offer(new User("Bob", "A1"));
q.offer(new User("Alice", "A2"));
q.offer(new User("Boss", "V1"));
 System.out.println(q.poll()); // Boss/V1
System.out.println(q.poll()); // Bob/A1
System.out.println(q.poll()); // Alice/A2
System.out.println(q.poll()); // null,因为队列为空


class UserComparator implements Comparator<User> {
    public int compare(User u1, User u2) {
        if (u1.number.charAt(0) == u2.number.charAt(0)) {
            // 如果两人的号都是A开头或者都是V开头,比较号的大小:
            return u1.number.compareTo(u2.number); // 先A1后A2 本来字典序也是返回-1，而负数代表不需要交换
        }
        if (u1.number.charAt(0) == 'V') {
            // u1的号码是V开头,优先级高:
            return -1; //返回-1 即为不需要交换位置，u1先于u2
        } else {
            return 1;
        }
    }
}

class User {
    public final String name;
    public final String number;

    public User(String name, String number) {
        this.name = name;
        this.number = number;
    }

    public String toString() {
        return name + "/" + number;
    }
}

    //String.compareTo

的返回值是整型，它是先比较对应字符的大小（ASCII码顺序），

以此类推，直至比较的字符或被比较的字符有一方全比较完，这时就比较字符的长度

    # 15.增强for循环

    for循环的一种，简化了数组和集合的遍历 只能获取集合元素。但是不能对集合进行操作 格式: for(元素数据类型 变量：数组或者Collection集合){xxx语句xxx} 弊端:增强for的目标不能是null 需要对增强for的目标先进行不为null的判断 优势:可以对空的数组或者集合进行循环

    for (TreeNode left : leftTrees) {
  for (TreeNode right : rightTrees) {
    TreeNode currTree = new TreeNode(i);
    currTree.left = left;
    currTree.right = right;
    allTrees.add(currTree);
  }
}
