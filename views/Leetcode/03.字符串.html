    # 500.键盘行问题

    // 判断数组是不是同一行字母打出来的
class Solution {
    static String one = "qwertyuiopQWERTYUIOP";  //static关键字不大写
    static String two = "asdfghjklASDFGHJKL";
    static String three = "zxcvbnmZXCVBNM";

    public String[] findWords(String[] words) {
        String [] ans = new String[words.length];  //创建String对象的方法，再[]中写数组长度
        int index = 0; //ans数组的idx索引
        for(String word: words) {  //foreach语句，循环变量要有变量类型
            int base = 0; //记录字符串s中字符属于第几行，初始化为0
            boolean flag = true;
            for(int i=0 ; i<word.length(); i++) {
                if(base == 0){base = parse(word.charAt(i));} //charAt()是根据索引获取到那个char
                else {
                    if(base != parse(word.charAt(i))){
                        flag = false;
                        break;
                    }
                }
            }
            if(flag) ans[index++] = word;  //在每个循环结束的时候根据flag判断是否添加这个单词word
        }
        return Arrays.copyOfRange(ans, 0, index);  //复制函数，这个函数要记住
    }

    public int parse(char x) {
        if(one.indexOf(x) != -1) return 1;
        else if(two.indexOf(x) != -1) return 2;
        else if(three.indexOf(x) != -1) return 3;
        else return -1;
    }
}

    # 520.检查大写字母

    检查一个单词是否符合大写规范

    class Solution {
    public boolean detectCapitalUse(String word) {
        if(word.length() == 1) return true;
        else {
            boolean base = parse(word.charAt(1)); //保存第二个字符的大小写，true为大写，为了最后判断2~末同小还是同大
            for(int i=2; i<word.length(); i++) {  //先看看2~末是否相同，不同直接false
                if(base != parse(word.charAt(i))) {
                    return false;  //与第二个字符不同则必定错误
                }
            }
            return base? parse(word.charAt(0)): true;
            //到这里说明第一位之后全都同大or同小
            //若2~末尾同大，则第一位必须大写
            //若2~末尾同小，则第一位可大可小，即恒为true
        }
    }
    public boolean parse(char c) {
        return 'A' <= c && c <= 'Z'; //true则为大写，否则为小写
    }
}

    # 541.反转字符串

    以2k为段反转前k长度的子串

    class Solution {
    public String reverseStr(String s, int k) {
        int n = s.length();
        char[] arr = s.toCharArray();  //String转换为char数组
        for(int i=0; i<n; i+=2*k) {  //每次都是reverse下一个2K长度子串位置，
            reverse(arr, i, Math.min(i+k, n)-1); //要么下一个k长度子串，要么截止到末尾的小于k长的字串
        }
        return new String(arr);
    }

    //这个方法reverse包括left也包括right
    public void reverse(char[] arr, int left, int right) {
        while(left<right) {
            char tmp = arr[right];
            arr[right] = arr[left];
            arr[left] = tmp;
            left++;
            right--;
        }
    }
}

    这道题int、String对象创建

List<String> ans = new ArrayList<String> ();

class Solution {
    public List<String> commonChars(String[] words) {
        //首先明确思路
        //方构造全局int数组minifreq长度为26，存储在每个word中都出现的字母的最小个数
        //意思说，如果某个单词中没有该字母，则该字母对应minifreq=0
        //循环结束，构造一个List = new ArrayList<String> 用add方法加入minifreq个某个小写字母
        int[] minifreq = new int[26]; //存储最小次数
        //Java中的fill方法在Arrays中
        Arrays.fill(minifreq, Integer.MAX_VALUE);
        //先填充超大，然后每次遍历都会得到3，2，1，0的时候保存住说明没有遇到
        //如果上来填0，岂不是上来就说有word没遇到这个char
        for(String word: words) {
            int[] freq = new int[26];
            for(int i=0;i<word.length();i++) {
                freq[word.charAt(i) - 'a']++;
            }
            for(int i=0; i<26;i++) {
                minifreq[i] = Math.min(minifreq[i], freq[i]);
            }
        }
        //处理完所有的word，得到了minifreq
        List<String> ans = new ArrayList<String>();
        for(int i=0;i<26;i++) {
            for(int j=0;j<minifreq[i] ;j++) {
                ans.add(String.valueOf((char)(i+'a')));  //添加到String数组中的必须是String，所以需要String.valueOf
            }
        }
        return ans;
    }
}

    # 1047. 删除字符串中的所有相邻重复项

    class Solution {
    public String removeDuplicates(String s) {
        StringBuffer ans = new StringBuffer();//新的对象StringBuffer，类似于队列，这里模拟栈
        int top = -1; //栈顶指示
        for(int i=0; i<s.length();i++) {
            if(top >= 0 && ans.charAt(top) == s.charAt(i)){  //必须在top>=0的时候才删除元素 安全
                ans.deleteCharAt(top--);  //和栈顶元素相同，删除栈顶元素，top下移
            }
            else {
                ans.append(s.charAt(i));  //将该和前一个不同的char入栈
                top++;
            }
        }
        return ans.toString();
    }
}

    # 14.最长公共前缀

    class Solution {
    public String longestCommonPrefix(String[] strs) {
        StringBuffer ans = new StringBuffer();
        int minlength = Integer.MAX_VALUE;
        for(String str: strs) {
            minlength = Math.min(minlength, str.length());
        }//得到需要遍历的长度
        for(int i=0;i<minlength;i++) {
            char tmp = strs[0].charAt(i);//当前需要检测的char
            for(String str:strs) {
                if(str.charAt(i) != tmp) return ans.toString();
            }
            ans.append(tmp);
        }
        return ans.toString();
    }
}

    class Solution {
    public int strStr(String haystack, String needle) {
        //思路就是指针去找neelde的第一个char，找到后开始匹配needle的每个char，通过则返回指针位置
        //否则继续，再否则返回-1
        if(needle.length()==0) {
            return 0;
        }
        char idx = needle.charAt(0); //待匹配首字母
        for(int i=0;i<haystack.length();i++) {
            if((haystack.length()-i)<needle.length()) {
                return -1;
            } //如果当前及之后的子串长度小于needle的长度，直接返回-1
            if(i >= haystack.length()) { //如果i已经超出了hay的长度则直接返回-1
                return -1;
            }
            if(haystack.charAt(i) == idx) {  //这里只需要判断正确情况即可，错误就continue
                int num = i; //保存当前位置 或许是答案
                boolean flag = true;
                for(int j=1; j<needle.length();j++) { //匹配剩下的length-1个char是否正确 这里已经从1开始了，length就不要-1了！
                    if(needle.charAt(j) == haystack.charAt(i+j)) {
                        continue;
                    }else {
                        flag = false;  //这里居然忘了写
                    }
                }//循环结束，可能是自然结束 可能是判断错误break
                if(flag == true) return num;
            }
        }
        return -1;
    }
}

    # 58. 最后一个单词的长度

    class Solution { //这题没什么好说的，就从后往前去找，可能最后还有若干空格要过滤掉，用flag判断是否出现过单词
    public int lengthOfLastWord(String s) {
        int len = s.length();
        boolean flag = false; //标志位，当检测到字母以后说明以有单词，置为true
        int res = 0;
        for(int i=len-1;i>=0;i--) {
            if(s.charAt(i) == ' ' && flag == false) continue; //继续查前面一个字母
            if(s.charAt(i) == ' ' && flag == true) return res;
            if(s.charAt(i) != ' ') {
                if(flag == false) flag = true;
                res++;
            }
        }
        return res;
    }
}

    # 125. 验证回文串

    class Solution {
    public boolean isPalindrome(String s) {
        StringBuffer buffer = new StringBuffer(); //过滤后的字母数字串
        //将原String过滤
        for(int i=0;i<s.length();i++) {
            char tmp = s.charAt(i);
            if(Character.isLetterOrDigit(tmp)) buffer.append(Character.toLowerCase(tmp));
        }
        StringBuffer reverse = new StringBuffer(buffer).reverse();
        return buffer.toString().equals(reverse.toString());
    }
}

    class Solution {
    public String convertToTitle(int columnNumber) {
        StringBuffer res = new StringBuffer();
        while(columnNumber != 0) {
            columnNumber --;
            res.append((char)('A' + columnNumber%26));
            columnNumber /= 26;
        }
        return res.reverse().toString();
    }
}

    # 205. 同构字符串

    这道题就是说双射问题，str a 中的char 需要映射到str b ，然后b也是这样 需要维护两张哈希表，用hashmap来实现，首先同时遍历两个string，如果出现x的映射不是y或者y的映射不是x，则false返回 如果没问题，则用新的映射put到哈希表中，循环外返回true

    class Solution {
    public boolean isIsomorphic(String s, String t) {
        //运用两个哈希表的方法
        Map<Character, Character> s2t = new HashMap<Character, Character>();
        Map<Character, Character> t2s = new HashMap<Character, Character>();
        for(int i=0;i<s.length();i++) {
            char x = s.charAt(i);
            char y = t.charAt(i);
            if((s2t.containsKey(x) && s2t.get(x) != y) || (t2s.containsKey(y) && t2s.get(y) != x)) {
                return false;
            }
            //如果不存在冲突，就向哈希表添加新的键值对(注意：key值相同的，value会被新的键值对替换)
            s2t.put(x, y);
            t2s.put(y, x);
        }
        return true;
    }
}

    # 242. 有效的字母异位词

    class Solution {
    public boolean isAnagram(String s, String t) {
        //思路非常简单，int数组长度26存储s中出现的次数，然后对t也维护一个表，最后比较两个表的对应位即可
        if(s.length() != t.length()) return false;
        int[] hash_s = new int[26];
        int[] hash_t = new int[26];
        Arrays.fill(hash_s, 0);
        Arrays.fill(hash_t, 0);
        for(int i=0;i<s.length();i++) {
            hash_s[s.charAt(i)-'a']++;
            hash_t[t.charAt(i)-'a']++;
        }
        for(int i=0;i<26;i++) {
            if(hash_s[i] != hash_t[i]) return false;
        }
        return true;
    }
}

    # 257. 二叉树的所有路径

    这道题用到深度优先搜索，递归函数编写，leecode给出了node的数据结构一看就懂 关键是递归函数的写法，弄清楚函数无条件返回的逻辑，而且递归函数要在最外层先判断是否当前根节点为null 另外，java语言会将传参实际修改，直接返回paths就可以了 写递归的时候注意逻辑，向当前path添加值放在第一步，然后当判断当前非叶子节点后直接在当前path的StringBuffer后面append一个"->"，方便了函数的递归调用 不可以在插值的时候加上"->3"这样，因为如果是从根出发，那最前面会多一个->，必须在判断非叶子之后加

    /**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<String> binaryTreePaths(TreeNode root) {
        //首先分析最后的输出应该是一个String类型的List列表
        //递归实现找路径，应该写一个递归函数，每次传一个treeNode类型的当前root，并把维护的答案List列表，以及当前路径的部分一起传下去
        List<String> paths = new ArrayList<String>();
        searchPath(root, "", paths);
        return paths;
    }//主函数逻辑
    public void searchPath(TreeNode root , String path, List<String> paths) {
        if(root != null) {  //函数必须在最外层判断非null，否则直接返回
            int currentVal = root.val;//应该先把这个当前根的val添加到当前路径中
            StringBuffer pathBuffer = new StringBuffer(path);
            pathBuffer.append(Integer.toString(currentVal)); //注意Int的toString方法是Integer类的静态方法调用
            if(root.left == null && root.right == null) {
                paths.add(pathBuffer.toString()); //添加到paths里即可，因为是void递归函数，也不需要返回什么
            } else {
                //不是叶子节点，先添加一个->
                pathBuffer.append("->");
                searchPath(root.left, pathBuffer.toString(), paths);
                searchPath(root.right, pathBuffer.toString(), paths);
            }
        }
    }
}

//然后，进入到searchPath(root.right..)，在这里走一步就发现是叶子，直接添加一条路，然后相当于最外层的search函数执行结束，程序返回paths
//看来java会以传参形式真正更新paths列表，最后返回它就可以了

    class Solution {
    public boolean wordPattern(String pattern, String s) {
        //一看就是哈希表,而且是双射
        Map<Character, String> ch2str = new HashMap<Character, String>();
        Map<String, Character> str2ch = new HashMap<String, Character>();
        int i=0;
        int j=0;//双指针方法去指示单词串s
        for(int k=0;k<pattern.length();k++) {
            if(i >= s.length()) return false; //如果循环到当前发现i的位置已经超过了s的长度，则必然错误
            char ch = pattern.charAt(k);
            j=i; //移动指针j移位
            while(j < s.length() && s.charAt(j) != ' ') j++; //找出当前单词位置,最后j是空格位
            String tmp = s.substring(i, j); //找出子串，不包含j位
            if((ch2str.containsKey(ch) && !ch2str.get(ch).equals(tmp)) || (str2ch.containsKey(tmp) && str2ch.get(tmp)!=ch)){
                return false;  //如果双射哈希表映射规律不对，则直接返回错误
            }
            ch2str.put(ch, tmp);
            str2ch.put(tmp, ch);
            i = j + 1;
        }
        return i >= s.length();
        //return true;  //不能这么写,因为我们前面是按照pattern的长度去检查的，可能单词串中还有别的单词，这种情况如果存在，那也是false
    }
}

    # 345. 反转字符串中的元音字母

    这道题相对也比较简单，思路上还是双指针 但要记忆String转char[] 的方法是char[] arr = s.toCharArray(); 以及char[] 转String的方法是new String(arr); 学习好的刷题方法，写几个工具函数 还有，String常量如 “aeiouAEIOU”可以直接用点调用String的一些方法,如.indexOf(char ch) 来判断常量字符串里是否由某个字符，没有则返回-1

    class Solution {
    public String reverseVowels(String s) {
        //还是双指针，但是指针的移动不再是++和--那么简单
        //而是需要去找到后/前一个原因字母的位置
        char[] arr = s.toCharArray();
        int i = 0;
        int j = s.length() - 1;
        while ( i < j) {
            //ij位置同时为元音字母则交换，否则ij把不是元音的找到
            if(isVowel(arr[i])&&isVowel(arr[j])) {
                swap(arr, i, j);
                i++;
                j--;
            }
            else {//不是同时是元音字母
                if(!isVowel(arr[i])) i++;
                if(!isVowel(arr[j])) j--;
            }
        }
        return new String(arr);
    }

    public boolean isVowel(char ch) {
        if("aeiouAEIOU".indexOf(ch) != -1) return true;  //可以直接使用常量字符串String的方法
        else return false;
    }

    public void swap(char[] arr, int i, int j) {
        char tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
}

    # 383. 赎金信

    这道题相对简单 就是判断s1是否能由s2中的字符写出来，s2字符只能使用一次，而且都是小写字母 那思路就是维护两个哈希表数出来s1都需要每个字母几次，然后数出来s2提供了每个字母多少个 如果s1需要的某个字母个数超过了s2中提供的该字母个数就返回false，就这么简单

    class Solution {
    public boolean canConstruct(String ransomNote, String magazine) {
        //哈希表即可
        int[] ransomHash = new int [26];
        int[] magazineHash = new int [26];
        Arrays.fill(ransomHash, 0);
        Arrays.fill(magazineHash, 0);
        for(int i=0;i<ransomNote.length();i++) {
            ransomHash[ransomNote.charAt(i)-'a']++; //得到ransom哈希表
        }
        for(int j=0;j<magazine.length();j++) {
            magazineHash[magazine.charAt(j)-'a']++; //得到maga哈希表
        }
        for(int k=0;k<26;k++) {
            if(ransomHash[k] > magazineHash[k]) return false;
        }
        return true;
    }
}

    # 387. 字符串中的第一个唯一字符

    返回的是索引，所以需要遍历第二次

    class Solution {
    public int firstUniqChar(String s) {
        //还是哈希表
        //先统计出26个字母出现的数目
        //然后第二次遍历s，找到第一个只出现了一次的字母，返回索引下标
        int[] hashtable = new int[26];
        Arrays.fill(hashtable, 0);
        for(int i=0;i<s.length();i++) {
            hashtable[s.charAt(i) - 'a']++;
        }
        for(int j=0;j<s.length();j++) {
            if(hashtable[s.charAt(j) - 'a'] == 1) return j;
        }
        return -1;
    }
}
