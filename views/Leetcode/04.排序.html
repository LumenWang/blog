    public int missingNumber(int[] nums) {
        int[] hsahtable = new int[nums.length + 1];
        Arrays.fill(hsahtable, 0);
        for(int i=0;i<nums.length;i++) {
            hsahtable[nums[i]]++;

    class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        //Map HashMap
        //Set HashSet 这道题使用
        //向hashset添加元素的方法是add
        //判断set中是否有某元素：contains()
        //获取set的长度 .size()
        Set<Integer> set1 = new HashSet<Integer>();
        Set<Integer> set2 = new HashSet<Integer>();
        for(int i=0;i<nums1.length;i++) {
            if(!set1.contains(nums1[i])) {
                set1.add(nums1[i]);
            }
        }
        for(int i=0;i<nums2.length;i++) {
            if(!set2.contains(nums2[i])) {
                set2.add(nums2[i]);
            }
        }
        return calcuIntersection(set1, set2);
    }
    public int[] calcuIntersection(Set<Integer> set1, Set<Integer> set2) {
        if(set1.size() > set2.size()) return calcuIntersection(set2, set1);
        Set<Integer> res_set = new HashSet<Integer>();
        for(int num: set1) {
            if(set2.contains(num)) res_set.add(num);
        }
        int [] res = new int[res_set.size()];
        int idx = 0;
        for(int num:res_set) {
            res[idx++] = num;
        }
        return res;
    }
}

    # 350. 两个数组的交集 II

    用到了HashMap取出每个数组中所有元素及其出现次数 然后遍历map1中所有出现元素，先找出交集元素，再求出两者出现次数的最小值 返回结果集用的是ArrayList，关于其API以及如何将其转化为int [] 在笔记里

    class Solution {
    public int[] intersect(int[] nums1, int[] nums2) {
        Map<Integer, Integer> map1 = new HashMap<Integer, Integer>();
        Map<Integer, Integer> map2 = new HashMap<Integer, Integer>();
        for(int i=0;i<nums1.length;i++) {
            if(!map1.containsKey(nums1[i])) {
                map1.put(nums1[i], 1);
            } else {
                map1.put(nums1[i], 1+map1.get(nums1[i]));
            }
        }
        for(int i=0;i<nums2.length;i++) {
            if(!map2.containsKey(nums2[i])) {
                map2.put(nums2[i], 1);
            } else {
                map2.put(nums2[i], 1+map2.get(nums2[i]));
            }
        }
        List<Integer> res = new ArrayList<Integer>();
        for(Map.Entry<Integer, Integer> entry: map1.entrySet()) {
            if(!map2.containsKey(entry.getKey())) continue;
            else {
                int num = entry.getKey();
                int count = Math.min(entry.getValue(), map2.get(entry.getKey()));
                for(int i=0;i<count;i++) {
                    res.add(num);
                }
            }
        }
        int[] res_final = new int[res.size()];
        for(int i=0;i<res.size();i++) {
            res_final[i] = res.get(i);
        }
        return res_final;
    }
}

    # 389. 找不同

    题意t比s多一个字母，有可能重复，让找出这个字母，但都是小写 用哈希表存储两个字符串出现的字母个数，然后比对

    class Solution {
    public char findTheDifference(String s, String t) {
        if(s.length() == 0 && t.length() == 1) return t.charAt(0);
        //hashtable
        int[] table1 = new int[26];
        int[] table2 = new int[26];
        for(int i=0;i<s.length();i++) {
            table1[s.charAt(i) - 'a']++;
        }
        for(int i=0;i<t.length();i++) {
            table2[t.charAt(i) - 'a']++;
        }
        for(int i=0;i<26;i++) {
            if(table1[i] == table2[i]) continue;
            else{
                return(char)('a' + i) ;
            }
        }
        return ' ';
    }
}

    # 414. 第三大的数

    这道题思路就是不要小看它，如果没有第三大则返回最大 数组可能出现重复，所以不能sort直接返回 先利用HashSet找到所有不重复元素，然后放回int数组中排序 如果超过三个则返回第三大的值，否则返回最大值

    class Solution {
    public int thirdMax(int[] nums) {
        Set<Integer> set = new HashSet<Integer>();
        for(int i=0;i<nums.length;i++) {
            set.add(nums[i]);
        }
        int[] list = new int[set.size()];
        int idx = 0;
        for(int elem: set) {
            list[idx++] = elem;
        }
        Arrays.sort(list);
        if(list.length < 3) return list[list.length - 1];
        else return list[list.length - 3];
    }
}

    # 455. 分发饼干

    这个题的想法就是排序之后用双指针，分别指向孩子和饼干 对孩子进行循环，后安排的孩子胃口一定比当前孩子大 为当前孩子找到比他的胃口刚好大一点的饼干，在j的索引不超过饼干数目且当前饼干不能满足孩子的情况下，不断j++ 停止循环的时候，如果j的索引已经越接，说明这个孩子在剩余饼干中无法被满足，那么再后面的更不会被满足，返回即可 如果停止循环是因为找到了合适的月饼，则j++、i++，继续处理下一个孩子 当正常跳出循环的时候，说明所有的孩子都处理完了，那么返回 i 即可

    class Solution {
    public int findContentChildren(int[] g, int[] s) {
        Arrays.sort(g);
        Arrays.sort(s);
        int i = 0; //孩子指针 最后输出应为i+1
        int j = 0; //饼干指针
        while(i<g.length) {//现在计划去满足i号孩子 后面的孩子都比这个孩子胃口大
            while( j<s.length && s[j]<g[i]) j++;
            if(j>=s.length) { //说明所有饼干都不再满足这个孩子
                return i;
            } else{
                j++; //将月饼给他
                i++; //去满足下一个孩子
            }
        }
        return i;
    }
}

    # 506. 相对名次

    核心就是计算每一个成绩对应的排名，前三名是字符串，后面的是排名数字对应的字符串 所以建立HashMap来存储分数和排名字符串的关系 然后最后再建立一个String[]来返回

    class Solution {
    public String[] findRelativeRanks(int[] score) {
        //首先score应该是不能动的，因为排名需要按照这个分数表来给出
        int[] score2 = score.clone();
        Arrays.sort(score2);
        Map<Integer, String> map = new HashMap<Integer, String>(); //存储分数对应的排名
        for(int i=score2.length-1;i>=0;i--) {
            if(i==score2.length-1) {
                map.put(score2[i], "Gold Medal");
            } else if(i==score2.length-2) {
                map.put(score2[i], "Silver Medal");
            } else if(i==score2.length-3) {
                map.put(score2[i], "Bronze Medal");
            } else {
                map.put(score2[i], score2.length-i+"");
            }
        } //得到所有分数和对应排名String
        String[] res = new String[score.length];
        for(int i=0;i<score.length;i++) {
            res[i] = map.get(score[i]);
        }
        return res;
    }
}

    # 561. 数组拆分 I

    这道题应该说比较简单，2n个数分为n组，求出min()的最大和。 想得到尽可能大的min()函数，则就需要把尽可能小的放在一起，大的放在一起，避免木桶效应 所以排序之后就是两两一对的组合

    class Solution {
    public int arrayPairSum(int[] nums) {
        //思路就是把尽可能小的两个数分到一起，排序即可
        Arrays.sort(nums);
        int len = nums.length;
        int res = 0;
        for(int i=0;i<len;i+=2) {
            res += nums[i];
        }
        return res;
    }
}

    # 594. 最长和谐子序列

    class Solution {
    public int findLHS(int[] nums) {
        //这道题可能不好想，实际上和谐子序列就是由两个数字组成的序列，相差最大为1
        //本来我以为子序列不能跳步，后来发现是可以的，只需要保证元素顺序就可以
        //所以方法就是先得到哈希映射，然后遍历哈希表，求出hash(x) + hash(x+1)中的最大值即为答案
        //所以这和子序列概念 屁关系没有
        //注意 不能用int一维数组作为哈希表，空间复杂度太大
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        for(int i=0;i<nums.length;i++) {
            if(!map.containsKey(nums[i])) {
                map.put(nums[i], 1);
            } else map.put(nums[i], map.get(nums[i])+1);
        }
        int res = 0;
        if(map.size() == 1) return 0;
        for(Map.Entry<Integer, Integer> entry : map.entrySet()) {
            if(!map.containsKey(entry.getKey()+1)) continue;
            int count = entry.getValue() + map.get(entry.getKey()+1);
            res = Math.max(res, count);
        }
        return res;
    }
}

    # 720. 词典中最长的单词

    这题是API不熟，怎样判断一个string是由给定的其他字符串逐步添加一个字母组成的，意思就是string的所有前缀都存在 怎样知道前缀都存在，那就是判断1，12，123等字符串前缀是否都存在 用到Set加快查询 String的子串API是substring(from, to) 如何比较两个string的字典序用到的是String.compareTo()方法

    class Solution {
    public String longestWord(String[] words) {
        //枚举String数组中的每一个String，如果它比当前的答案更长或者等长却字典序更小，且它的所有前缀都存在，则更新为新答案
        String ans = "";
        Set<String> set = new HashSet<String>();
        for(String word: words) {
            set.add(word);
        } //将所有单词加入Set加快查询效率
        for(String word: words) {
            boolean flag = false;
            if(word.length() > ans.length() || (word.length()==ans.length() && word.compareTo(ans)<0 )) {
                flag = true;
                for(int k=1;k<word.length();k++) {
                    if(!set.contains(word.substring(0, k))) {
                        flag = false;
                        break;
                    }
                }
                if(flag == true) {
                    ans = word;
                }
            }
        }
        return ans;
    }
}

    # 747. 至少是其他数字两倍的最大数

    这道题的话就很简单，先用哈希表存储每个数的下标 然后排序，最后一位就是最大元素 如果数组只一个数，返回0 否则，就拿出倒1和倒2的数试一下，然后返回哈希表中存的下标即可

    class Solution {
    public int dominantIndex(int[] nums) {
        Map<Integer, Integer> map = new HashMap<Integer, Integer>();
        for(int i=0;i<nums.length;i++) {
            map.put(nums[i], i); //kv对存的是数字与其对应的下标
        }
        Arrays.sort(nums);
        if(nums.length == 1) {
            return 0;
        }
        else {
            int max_num = nums[nums.length - 1];
            int sec_num = nums[nums.length - 2];
            if(max_num >= 2*sec_num) return map.get(max_num);
            else return -1;
        }
    }
}

    ###888. 公平的糖果棒交换 题意是：交换两个int数组中的一个值，使两个数组的元素和相等，返回两个数组分别交换的那个元素，一定有答案 思路就是，首先哈希表保存每个数组中的元素 然后计算当前数组和之差，然后该值取半就是需要交换的两个数之差 巧妙利用正负值 然后从数组a中遍历每个元素，看看满足交换条件的元素在数组b中是否存在，如果有就返回即可

    class Solution {
    public int[] fairCandySwap(int[] aliceSizes, int[] bobSizes) {
        int[] ans = new int[2];
        Map<Integer, Integer> mapa = new HashMap<Integer, Integer>();
        Map<Integer, Integer> mapb = new HashMap<Integer, Integer>();
        for(int i=0;i<aliceSizes.length;i++) mapa.put(aliceSizes[i], 1);
        for(int i=0;i<bobSizes.length;i++) mapb.put(bobSizes[i], 1);

        int standard = calcu(aliceSizes) - calcu(bobSizes); //如果是正的，则需要ali得到bob更小的
        int aver = standard / 2;
        for(int i=0;i<aliceSizes.length;i++) {
            if(mapb.containsKey(aliceSizes[i]-aver)) {
                ans[0] = aliceSizes[i];
                ans[1] = aliceSizes[i] - aver;
                return ans;
            }
        }
        return ans;
    }
    public int calcu(int[] arr) {
        int res = 0;
        for(int i=0;i<arr.length;i++) {
            res += arr[i];
        }
        return res;
    }
}

    # 905. 按奇偶排序数组

    这道题也没什么好说的，主要是先遍历得到偶数就加入ans中，奇数先保存倒一个list里，然后再加到ans里 ArrayList的API还不是很熟

    class Solution {
    public int[] sortArrayByParity(int[] nums) {
        List<Integer> list = new ArrayList<Integer>();
        int[] ans = new int[nums.length];
        int idx = 0;
        for(int i=0;i<nums.length;i++) {
            if(nums[i] % 2 == 0) ans[idx++] = nums[i];
            else list.add(nums[i]);
        }
        for(int i=0;i<list.size();i++) {
            ans[idx++] = list.get(i);
        }
        return ans;
    }
}
