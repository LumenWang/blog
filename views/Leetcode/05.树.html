    # 94. 二叉树的中序遍历

    class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        // left middle right
        //先判断是否左孩子为空，进入递归左子树。如果为空则向答案中添加当前的val，然后遍历右子树，最后返回list
        List<Integer> list = new ArrayList<Integer>();
        if(root == null) return list;
        else if(root != null && root.left==null && root.right==null) {
            list.add(root.val);
            return list;
        }
        else { //根节点不为空，且不是独立的叶子，即为有子树
            middleSearch(root, list);
        }
        return list;
    }
    public void middleSearch(TreeNode root, List<Integer> list) {
        if(root.left != null) middleSearch(root.left, list);
        list.add(root.val);
        if(root.right != null) middleSearch(root.right, list);
    }
}

    # 100. 相同的树

    判断两个数是否在数值和结构上都完全相同 本来想递归找到两个树的中序遍历结果，后来发现其实是不行的 [如果同时找出中序和根序再比对呢？] 最简单是深度优先递归，每一步都同步判断两棵树的情况即可

    class Solution {
    public boolean isSameTree(TreeNode p, TreeNode q) {
        //中序遍历其实也无法唯一确定树，只能递归深度优先的方式同步匹配两颗树的所有同位根、同位左子树和同位右子树
        if(p==null && q==null) return true;
        else if(p==null || q==null) return false;
        else if(p.val != q.val) return false;
        else { //两棵树都非空且当前根的值也相等
        return isSameTree(p.left, q.left) && isSameTree(p.right, q.right);
        }
    }
}

    # 101. 对称二叉树

    看来递归的思想还是不太懂 用镜像指针

    class Solution {
    public boolean isSymmetric(TreeNode root) {
        //镜像对称问题用镜像指针解决
        return check(root, root);
    }
    public boolean check(TreeNode p, TreeNode q) {
        //全null，一个null，都不为null
        if(p==null && q==null) return true; //全空视为相同
        else if(p==null || q==null) return false;
        else {
            return p.val==q.val && check(p.left, q.right) && check(p.right, q.left);
        }
    }
}

    # 104. 二叉树的最大深度

    第一种方法是深度优先搜索，用递归调用的方式不断地计算1+子树深度

    class Solution {
    public int maxDepth(TreeNode root) {
        //递归, 深度优先搜索
        if(root == null) return 0;
        else {
            int left = maxDepth(root.left);
            int right = maxDepth(root.right);
            return Math.max(left, right) + 1; //走到这里如果是叶子，则该叶子的深度为1，再向上返回
        }
    }
}

    第二种方法 广度优先搜索，用队列一层一层存储树的元素。每次存下一层元素之前需要把这层元素一个个弹出并将其左右孩子加入队列尾部，所以循环终止条件就是队列为空

    class Solution {
    public int maxDepth(TreeNode root) {
        // 一行一行添加树中的元素，用linkedlist添加
        // 第一次肯定只有一个根，每次需要保证把这一行中的元素全部依次取出，并将这些元素的左右孩子加入到链表的尾部
        if(root == null) return 0;
        LinkedList<TreeNode> linkedlist = new LinkedList<TreeNode>();  //创建一个链表代表队列
        linkedlist.add(root);//先把root根加入进去
        int ans = 0; //树的层数
        while(!linkedlist.isEmpty()) {  //对于每一个元素应该都弹出并检查其左右孩子，所以最终队列里没有元素了
            //而且，在同一个状态下应该保证队列里只有同一层的孩子
            int size = linkedlist.size();
            while(size>0) {
                TreeNode tmp = linkedlist.poll();
                if(tmp.left != null) linkedlist.add(tmp.left);
                if(tmp.right != null) linkedlist.add(tmp.right);
                size--;
            }//处理完了一层的元素，将这一层的意义添加
            ans++;
        }
        return ans;
    }
}

    # 108. 将有序数组转换为二叉搜索树

    这道题也很重要，用递归的方式去建立一个二叉搜索树
首先根的选择肯定是数组的mid位元素
然后，左右分治递归，去建立左右子树，注意left和right
如果left》right则return null
否则，先创建一个节点，将值赋值到节点上，然后再递归创建左右子树
class Solution {
    public TreeNode sortedArrayToBST(int[] nums) {
        //考虑也是递归的方法去建树
        return builder(nums, 0, nums.length-1);
    }
    public TreeNode builder(int[] nums, int left, int right) {
        if(left > right) return null;
        int mid = (left + right)/2;
        TreeNode root = new TreeNode(nums[mid]);
        root.left = builder(nums, left, mid-1);
        root.right = builder(nums, mid+1, right);
        return root;
    }
}

    # 110. 平衡二叉树

    //这道题就是给一个二叉树，判断是不是平衡二叉树
//大思路，就是递归判断每一个节点的左右孩子树的深度差是否大于1，
//所以，需要一个计算root树深度的函数，方法还是递归
//然后回到主函数，利用这个函数去判断，方法也是递归
class Solution {
    public boolean isBalanced(TreeNode root) {
        if(root==null) return true; //如果为null节点，返回true 也是递归的最深处条件
        else { //如果不是null节点
            if(Math.abs(height(root.left)-height(root.right))>1) return false; //如果不满足，则直接返回false
            else return isBalanced(root.left) && isBalanced(root.right); //否则，返回左右子树都是平衡二叉树
        }
    }
    public int height(TreeNode root) { //递归计算一个节点的深度
        if(root == null) return 0; //必要，递归的最深处条件
        else return Math.max(height(root.left), height(root.right)) + 1;
    }
    //重点是一个计算孩子树的高度的函数
}

    # 111. 二叉树的最小深度

    这道题本来想偷懒结果错了 一定要想清楚[2,null,3,null,4,null,5,null,6] 这种情况 你计算最小深度思路是没错，就是找一个最小深度的叶子返回其深度就行了 但是，不能计算一个空的左、右子树的深度(按照递归去计算那就是0，然后递归返回就是1)，因为那不是叶子 所以去递归计算左右子树的最小深度的时候要先判断是否为null 用一个变量去存储对于当前root来说最小的depth

    class Solution {
    public int minDepth(TreeNode root) {
        //左右子树中较小深度+1，就返回了当前节点最小深度，最后返回root的这个值
        if(root==null) return 0;
        if(root.left == null && root.right == null) return 1;

        int max_depth = Integer.MAX_VALUE;

        // return Math.min(minDepth(root.left), minDepth(root.right)) + 1;
        //这样写的错误就是 是null的子树位置不应该考虑其作为最小深度，否则如[2,null,3,null,4,null,5,null,6] 就只能返回1
        if(root.left != null) { //只有不是空的子树才去计算它的深度
            max_depth = Math.min(max_depth, minDepth(root.left));
        }
        if(root.right != null) {
            max_depth = Math.min(max_depth, minDepth(root.right));
        }
        return max_depth + 1;
    }
}

    # 112. 路径总和

    这道题给你一个值，让你验证树中是否有到叶子节点的路径和为该值 方法就是，先写一个函数，用来递归的去计算每一个叶子节点的所有可能的路径和 方法就是在递归函数中传引用currSum代表当前路径和，一个list代表已知答案集合 如果一个树是空的则不做任何操作 如果是叶子节点，则将currSUm+叶子的val然后存入list即可 如果不是叶子也非空，则需要判断左右子树，然后分别调用递归函数本身，此时currsum参数+=root.val这个中间节点的值 最后遍历list看看有无target即可 这道题是自己写出来的

    class Solution {
    public boolean hasPathSum(TreeNode root, int targetSum) {
        ArrayList<Integer> ansList = new ArrayList<Integer>();
        calcuSum(root, 0, ansList);
        for(int sum: ansList) {
            if(sum==targetSum) return true;
        }
        return false;
    }

    public void calcuSum(TreeNode root, int currSum, ArrayList<Integer> ansList) {
        if(root==null) {}
        else if(root.left==null && root.right==null){  //叶子节点
            currSum += root.val;
            ansList.add(currSum);
        } else {  //非叶子节点
            if(root.left!=null) {
                calcuSum(root.left, currSum+root.val, ansList);
            }
            if(root.right!=null) {
                calcuSum(root.right, currSum+root.val, ansList);
            }
        }
    }
}

    # 144. 二叉树的前序遍历

    这道题就是给你一个root节点 返回前序遍历 java很好的一点就是工具函数传引用 自己写前序遍历递归函数，参数中有一个引用list存储答案 在当前root非空前提下，先加入root.val，然后判断左右，分别递归调用，如果为空就不调用也不添加 这道题是自己写出来的

    class Solution {  //root left right
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> ansList = new ArrayList<Integer>();
        frontOrderTraverse(root, ansList);
        return ansList;
    }
    public void frontOrderTraverse(TreeNode root, List<Integer> ansList){
        if(root!=null) {
            ansList.add(root.val);
            if(root.left!=null) {
                frontOrderTraverse(root.left, ansList);
            }
            if(root.right!=null) {
                frontOrderTraverse(root.right, ansList);
            }
        }
    }
}

    # 226. 翻转二叉树

    这道题其实不是镜像指针也不是BFS，因为都不方便 思路就是，对于每一个节点，如果非空非叶，则先把left和right整体交换，然后再对left和right分别递归处理 如果root是叶子，则返回它本身 如果不是，则按上述处理 对每一个节点都是先交换左右，再递归到左右

    class Solution {
    public TreeNode invertTree(TreeNode root) {
        if(root!=null) {
            if(root.left==null && root.right==null) { //叶子节点，返回自己本身节点
                return root;
            }
            else { //不是空，也不是叶子节点，需要先反转当前的左右孩子，再递归到左右孩子
                TreeNode tmp = root.left;
                root.left = root.right;
                root.right = tmp;
                if(root.left!=null) root.left = invertTree(root.left);
                if(root.right!=null) root.right = invertTree(root.right);
            }
            return root;  //每次递归都是要返回当前根节点的
        }
        return null;
    }
}

        TreeNode node = root;
        while(node != null) {
            if(p.val<node.val && q.val<node.val) {
                node = node.left;
            }
            else if(p.val>node.val && q.val>node.val) {
                node = node.right;
            }
            else return node;
        }
        return node;
    }
}

    # 404. 左叶子之和

    递归深度优先搜索计算所有左孩子的累加和 在递归函数中创建临时变量，最后返回的是最外层dfs的ans变量，内部每一层调用都返回那一层的ans变量 每一层递归，函数都新建一个ans变量，初始为0 如果 left 存在且为叶子则累加，如果left存在但不是叶子则累加其dfs函数返回值 如果 right 存在且不为叶子，则累加right节点的dfs函数返回值 其实最深层的dfs函数是返回了新建的ans变量且是0，因为没有left也没有right了，然后一层层返回

    class Solution {  //dfs方法
    public int sumOfLeftLeaves(TreeNode root) {
        //遍历树中的每一个节点，如果左孩子是叶子节点，则累加到结果中
        //深度优先搜素
        return (root==null)?0:dfs(root);
    }
    public int dfs(TreeNode root) {
        int ans = 0;
        if(root.left!=null) {
            ans += isLeaf(root.left)?root.left.val:dfs(root.left);
        }
        if(root.right!=null && !isLeaf(root.right)) {
            ans += dfs(root.right);
        }
        return ans;
    }
    public boolean isLeaf(TreeNode root) {
        if(root==null) return false;
        else if(root.left==null && root.right==null) return true;
        else return false;
    }
}

//方法2 广度优先 一个个node处理，如果left为叶子，则累加，否则入队列(非空),right不是叶子则入队列
class Solution{
    public int sumOfLeftLeaves(TreeNode root) {
        if(root==null) return 0;
        LinkedList<TreeNode> queue = new LinkedList<TreeNode>();
        int ans = 0;
        queue.add(root);
        while(!queue.isEmpty()) {  //一个个处理即可
            TreeNode tmp = queue.pop();
            if(tmp.left != null) {
                if(tmp.left.left==null && tmp.left.right==null) {  //如果left是叶子节点就累加left
                    ans += tmp.left.val;
                } else {
                    queue.add(tmp.left); //left不是叶子节点就入队列
                }
            }
            if(tmp.right!=null) {
                if(tmp.right.left != null || tmp.right.right != null) {
                    queue.add(tmp.right);
                }
            }
        }
        return ans;
    }
}

    # 501. 二叉搜索树中的众数

    这道题又给了我很多新启发，比如java的刷题class里可以定义一些全局变量 这道题的思路就是中序遍历这棵树，得到递增序列 找众数，利用这个递增序列的性质，创建几个变量 保存当前递增序列的元素curr，该元素的当前计数cur_count，目前找到的众数的最大出现次数max_count 然后利用深度优先 左 根 右 的次序去遍历每个node，找到其val值 如果当前val值和递增序列的上一个元素curr相等，则cur_count++ 如果当前val值不再等于curr，则将curr替换为当前val，并且cur_count = 1重置 如果cur_count值超过了max_count，则清空掉之前找到的所有众数，并将当前新众数添加进去 其实，相当于在dfs中处理当前node的时候（左：递归左孩子，根：执行parse，右：递归右孩子）去执行parse函数去做上面这些比较和赋值工作、添加找出的众数等工作

    class Solution {
    int curr, cur_count;
    int max_count = 0;
    List<Integer> list = new ArrayList<Integer>();

    public int[] findMode(TreeNode root) {
        dfs(root);
        int[] ans = new int[list.size()];
        for(int i=0;i<list.size();i++) {
            ans[i] = list.get(i);
        }
        return ans;
    }

    public void dfs(TreeNode node) {  //中序遍历 左 根 右
        if(node == null){return;}
        dfs(node.left);
        parse(node.val);
        dfs(node.right);
    }

    public void parse(int val) { //对于每一个遍历到的node的值的处理
        if(curr==val) cur_count++; //如果当前node值也等于curr保存的数，则计数++
        if(curr!=val) {//如果不相等了，则curr设置为当前node的val，然后curr_count置为1
            curr = val;
            cur_count = 1;
        }
        if(cur_count == max_count) list.add(curr); //如果当前计数等于以往众数，则添加候选众数
        if(cur_count > max_count) { //如果当前元素的次数已经超过了之前的众数，则设置max_count，清空之前的假众数，添加新的众数
            max_count = cur_count;
            list.clear();
            list.add(curr);
        }
    }
}

    # 530. 二叉搜索树的最小绝对差

    这道题和783. 二叉搜索树节点最小距离是一样的 这道题是计算树中任意两个节点的差的绝对值的最小值 既然是求最小值，那么这个值肯定是中序遍历序列中连续的两个元素的差的绝对值，因为中序是递增序列，相邻元素差的绝对值肯定是最小的 所以写一个dfs函数，将当前node的val和上一次的元素对比，如果这个差的绝对值比上一个小，则更新 写法上使用了上一题的写法

    class Solution {
    int pre = Integer.MAX_VALUE;
    int ans = Integer.MAX_VALUE; //保存那个差的绝对值
    public int getMinimumDifference(TreeNode root) {
        dfs(root);
        return ans;
    }
    public void dfs(TreeNode node) {
        if(node == null) return;
        dfs(node.left);
        ans = Math.min(Math.abs(node.val-pre), ans);
        pre = node.val;
        dfs(node.right);
    }
}

    class Solution {
    public boolean isSubtree(TreeNode root, TreeNode subRoot) {
        //肯定是要遍历root这棵树的
        if(subRoot==null) return true;
        else { //sub不为空树
            if(root==null) return false;
            else {  //都不为空

                while(!list.isEmpty()) {

                    if(tmp.left!=null) list.add(tmp.left);
                    if(tmp.right!=null) list.add(tmp.right);
                    if(isSameTree(tmp, subRoot)) return true;
                }
                return false;
            }
        }
    }

        else if(root2==null && root1!=null) return false;
        else { //都不为空
            if(root1.val!=root2.val) return false;
            else return isSameTree(root1.left, root2.left)&&isSameTree(root1.right, root2.right);
        }
    }
}

        buffer.append(tree2String(node.left));

                if(tmp.right!=null) list.addFirst(tmp.right);

        }
        return ans;
    }

    # 671. 二叉树中第二小的节点

    这道题给出一个特殊的二叉树，任意一个node的孩子数为0或2，且node.val = min(left, right) 要你输出树里第二小的节点值 通过题意可以知道root节点值肯定是最小的，如何去找到第二小 方法就是递归，首先任何一个绝对大于root节点值的val都可能是第二小的元素 但是，如果我们当前已经保存了一个ans，且当前node.val比它打或者等于它，那么node.val都不是ans 且，一个node的val肯定是以它为根的树中的最小值，所以这种情况不必递归其left和right 所以在dfs函数中 基本顺序是根 左 右

    class Solution {
    int ans = -1;
    int rootValue;
    public int findSecondMinimumValue(TreeNode root) {
        //利用好这个二叉树的性质，每个节点的值都等于min(left, right)
        //所以如果当前节点我们知道它过大，那么其左右孩子树都不需要再去遍历
        //每一个大于根节点值的val都有可能是这个第二小的元素，但前提是目前已知ans不为-1，且这个元素比之前找到的ans更小
        rootValue = root.val;
        dfs(root);
        return ans;
    }
    public void dfs(TreeNode node) {
        //应该是根 左 右 的顺序
        if(node==null) return;
        if(ans!=-1 && node.val>=ans) return; //已找到了一个ans，当前值比它大 绝对不可能是第二小的元素
        if(node.val>rootValue) {
            ans = node.val;
        }
        //dfs递归不应该在上一个if中
        dfs(node.left);
        dfs(node.right);
    }
}

    # 559. N 叉树的最大深度

    这道题就是利用n叉树节点的children是一个 list 
dfs方法中，递归去遍历，如果是null或者叶子则返回0和1
如果不是叶子，需要返回1+max_child_depth
这个最大值需要遍历所有的子节点去计算其深度并得到最大值
利用到了list.sort(comparator<>(){})接口的实现
class Solution {
    public int maxDepth(Node root) {
        if(root==null) return 0;
        if(root.children.size()==0) return 1;
        else {
            List<Integer> childrenDepth = new ArrayList<Integer>();
            for(int i=0;i<root.children.size();i++) {
                childrenDepth.add(maxDepth(root.children.get(i)));
            }
            childrenDepth.sort(new Comparator<Integer>(){
                public int compare(Integer o1, Integer o2) {
                    int res = o1>o2?1:-1;
                    return res;
                }
            });
            return 1+childrenDepth.get(childrenDepth.size()-1);
        }
    }
}

    class KthLargest {
    int k_largest;
    PriorityQueue<Integer> priority_queue;

    public KthLargest(int k, int[] nums) {
        this.k_largest = k;
        this.priority_queue = new PriorityQueue<Integer>();
        for(int i=0;i<nums.length;i++) {
            add(nums[i]); //这里不能调用PriorityQueue的add方法，因为优先队列必须一直保持只有k个数 其poll出来的才是第k大的数
        }
    }
    
    public int add(int val) {  //插入并返回第k大的元素
        priority_queue.offer(val);
        if(priority_queue.size()>k_largest) priority_queue.poll();
        return priority_queue.peek(); //返回队首元素 但不能删除
    }
}

    class Solution {

        }

    }
    public void inorderTraverse(TreeNode node) {
        if(node==null) return;
        //左 根 右
        inorderTraverse(node.left);
        list.addFirst(node.val);
        inorderTraverse(node.right);
    }
}

    //这道题很好 考到了BFS遍历 两个队列一个存node一个记录node的深度
 //还用到了基本的parent节点的存储 depth的存储等
class Solution {
    //x和y对应节点信息
    int x_val;
    int x_depth;
    TreeNode x_parent;
    boolean x_founded = false;

    int y_val;
    int y_depth;
    TreeNode y_parent;
    boolean y_founded = false;

    public boolean isCousins(TreeNode root, int x, int y) {
        this.x_val = x;
        this.y_val = y;
        LinkedList<TreeNode> nodeList = new LinkedList<TreeNode>();
        LinkedList<Integer> depthList = new LinkedList<Integer>();
        nodeList.addFirst(root);
        depthList.addFirst(0);
        checkNode(root, null, 0); //首先检查根节点 万一是x或y之一呢
        while(!nodeList.isEmpty()) {
            TreeNode tmpNode = nodeList.removeLast();
            int tmpDepth = depthList.removeLast();
            if(tmpNode.left!=null) {
                checkNode(tmpNode.left, tmpNode, tmpDepth+1);
                nodeList.addFirst(tmpNode.left);
                depthList.addFirst(tmpDepth+1);
            }
            if(tmpNode.right!=null) {
                checkNode(tmpNode.right, tmpNode, tmpDepth+1);
                nodeList.addFirst(tmpNode.right);
                depthList.addFirst(tmpDepth+1);
            }
        }
        if(this.x_founded && this.y_founded){
            return x_depth==y_depth && x_parent!=y_parent;
        } else return false;
    }

    public void checkNode(TreeNode node, TreeNode parent, int depth) {
        if(node.val==this.x_val) {
            this.x_founded = true;
            this.x_depth = depth;
            this.x_parent = parent;
        }
        if(node.val==this.y_val) {
            this.y_founded = true;
            this.y_depth = depth;
            this.y_parent = parent;
        }
    }
}

    # 1022. 从根到叶的二进制数之和

    这道题的话是计算一个二进制树中，每一条到根节点的路径上的node.val组成的二进制数转化为10进制之后的总和
方法是dfs，在遍历到每一个节点的时候传参为该node和到该结点之前所得到的二进制字符串
如果当前遍历到的node是叶子，那么要将处理完的String添加到String类型的list，待处理
遍历完所有节点，处理List<String> 转化二进制字符串为int并求和
class Solution {
    int res = 0; //存储最后的32位整数
    List<String> binaryStrList = new ArrayList<String>();

    public int sumRootToLeaf(TreeNode root) {
        if(root.left==null && root.right==null) return root.val;
        dfs(root, new StringBuffer());
        for(int i=0;i<binaryStrList.size();i++) {
            String tmp = binaryStrList.get(i);
            int len = tmp.length();
            for(int j=len-1; j>=0; j--) { 
                res += Math.pow(2, j) * (tmp.charAt(len-j-1) - '0');
            }
        }
        return res;
    }

    public void dfs(TreeNode node, StringBuffer buffer) { //到该node之前的string
        StringBuffer buffer_tmp = new StringBuffer(buffer);
        buffer_tmp.append(node.val+"");  //将当前node的值添加到二进制串中
        if(node.left==null && node.right==null) { //node是叶子节点
            binaryStrList.add(buffer_tmp.toString());
        } else { //不是叶子节点
            if(node.left!=null) {
                dfs(node.left, buffer_tmp);
            }
            if(node.right!=null) {
                dfs(node.right, buffer_tmp);
            }
        }
    }
}

    # 面试题 04.02. 最小高度树

    //这道题的关键在于想到用二分法递归建立子树
//关键要想清楚left和right middle是nums的索引下标
//因为我的middle是直接取整，所以当有两个待选middle的时候总是选择那个小的
//注意，当middle==0的时候，left应该是null
class Solution {
    int[] nums;
    public TreeNode sortedArrayToBST(int[] nums) {
        this.nums = nums;
        if(nums.length==0) return null;
        return binaryCreateTree(0, nums.length-1); //二分法递归建树
    }
    public TreeNode binaryCreateTree(int left, int right) { //初始化调用是0 length-1
        int middle = (left+right) / 2;  //计算对于目前数组来说作为根节点的的数的下标
        TreeNode node = new TreeNode(nums[middle]);
        if(left<=middle-1 && middle!=0) node.left = binaryCreateTree(left, middle-1);
        if(right>=middle+1 && middle!=nums.length-1) node.right = binaryCreateTree(middle+1, right);
        return node;
    }
}

    # 面试题 04.04. 检查平衡性

    这道题好像做过 就是检查二叉树的左右子树的高度差不大于1 需要遍历root中的所有节点，并验证每个节点的左右孩子子树都是平衡的 不需要用什么BFS和队列存储所有节点 只需要递归调用主函数，先验证root节点的平衡性，再递归到其left和right孩子 最后返回的是一个boolean值，需要保证当前root节点是平衡节点的前提下，isBalanced(root.left) && isBalanced(root.right) 前面的一些基础处理: 1.如果为空节点 直接返回true 2.如果是叶子节点 直接返回true 3.如果当前节点的左右孩子树的高度差大于1 直接返回false 4.如果以上都处理完了，那就返回isBalanced(root.left) && isBalanced(root.right)

    class Solution {
    public boolean isBalanced(TreeNode root) {
        if(root==null) return true;
        if(root.left==null && root.right==null) return true;  //如果是叶子节点也返回true
        if(Math.abs(calcuHeight(root.left)-calcuHeight(root.right))>1) return false;
        else {
            return isBalanced(root.left) && isBalanced(root.right);
        }
    }

    public int calcuHeight(TreeNode node) { //计算以node为根的树的深度(根节点也算1层)
        if(node==null) return 0;
        if(node.left==null && node.right==null) return 1;
        return 1 + Math.max(calcuHeight(node.left), calcuHeight(node.right));
    }
}
