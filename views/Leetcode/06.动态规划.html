    # 70. 爬楼梯

    滚动数组的思想刚开始接触，因为f(n) = f(n-1) + f(n-2) 只与之前的两个计算值相关，所以只需要保存这两个值即可

    //滚动数组+动态规划 p q r 分别存储 f(n-2) f(n-1) f(n)
class Solution {
    public int climbStairs(int n) {
        int p=0, q=0, r=1; //初始化为0 0 1
        for(int i=1;i<=n;i++) {  //控制循环次数为传参数n
            p = q;
            q = r;
            r = p+q;
        }
        return r;
    }
}

        List<List<Integer>> res = new ArrayList<>();
        if(numRows==0) return res;

    }
}

    # 121. 买卖股票的最佳时机

    //这道题只关心我最多能赚到多少钱，不需要记住买卖时间
//所以我只遍历数组一次，记录价格最低点，然后不断地考虑如果当天卖出能赚多少钱
//价格最低点的初始值是max_value
class Solution {
    public int maxProfit(int[] prices) {
        int min_value = Integer.MAX_VALUE;
        int max_profit = 0;
        for(int i=0;i<prices.length;i++){
            if(prices[i]<min_value) min_value = prices[i]; //当前价格为最低
            else { //如果当前价格不是历史最低 那么今天就可以得到一个利润
                int current_profit = prices[i] - min_value;
                if(max_profit<current_profit) max_profit = current_profit;
            }
        }
        return max_profit;
    }
}

    # 338. 比特位计数

    克宁汉算法，x&(x-1)的次数就是x二进制表示中1的个数

    class Solution {
    public int[] countBits(int n) {
        int[] bitcounts = new int[n+1];
        for(int i=0;i<=n;i++) {
            bitcounts[i] = calcuBitCounts(i);
        }
        return bitcounts;
    }
    private int calcuBitCounts(int n) {
        int counts = 0;
        while(n>0) {
            n = n & (n-1);
            counts++;
        }
        return counts;
    }
}

    # 392. 判断子序列

    双指针，用指针s依次处理s中的每个字符，用指针t去检测 如果找到了对应的字符，即时停止内层循环和外层循环，t指针也要++ 如果找不到，则直接返回false

    class Solution {
    public boolean isSubsequence(String s, String t) {
        int s_idx = 0;
        int t_idx = 0;
        while(s_idx<s.length()) {
            char tmp = s.charAt(s_idx);
            boolean have = false;
            while(t_idx<t.length()) {
                if(t.charAt(t_idx)!=tmp) {
                    t_idx++;
                } else {
                    have = true;
                    t_idx++;
                    break;
                }
            }
            if(have==false) return false;
            s_idx++;
        }
        return true;
    }
}

    # 746. 使用最小花费爬楼梯

    滚动数组 + 动态规划 的思想，dp[0] = dp[1] = 0 注意循环的终止次数

    //对于计算中间的某一个阶梯i，到达它的方法只有1.从i-1过来 2.从i-2过来
//所以到达阶梯i的最低花费dp[i] = min(dp[i-1]+cost[i-1], dp[i-2]+cost[i-2]);
//属于动态规划问题
//题目说明了是可以从下标0或者下标1作为起始元素，所以说达到dp[0]和dp[1]的最低花费应该都是0
//dp[2] = min(dp[1]+cost[1], dp[0]+cost[0]); //2号台阶可以从0走两步过来，或者从1走一步过来 dp[0]/dp[1]都为0
//还是滚动数组的思想 因为dp[i]只与dp[i-1]和dp[i-2]有关 所以arr[0] arr[1] arr[2] 分别代表dp[i-2] dp[i-1] dp[i]
//注意这道题的题意，走台阶到楼顶意味着cost的最后一个元素值也会被用上，所以循环应该到i<=cost.length
class Solution {
    public int minCostClimbingStairs(int[] cost) {
        int[] arr = new int[3];
        arr[0] = 0;
        arr[1] = 0;
        for(int i=2; i<=cost.length; i++) {  //cost代表的是台阶代价，所以到达楼顶意味着可能把最后一阶台阶的cost也用上
            arr[2] = Math.min(arr[0]+cost[i-2], arr[1]+cost[i-1]); //arr[2] 保存的永远是新计算的值
            arr[0] = arr[1]; //计算后紧接着要把数据向前保存
            arr[1] = arr[2]; 
        }
        return arr[1]; //返回的应该是arr[1]
    }
}

    # 509. 斐波那契数

    class Solution {
    public int fib(int n) {
        if(n==0) return 0;
        int[] arr = new int[3];
        arr[0] = 0;
        arr[1] = 1;
        for(int i=2; i<=n; i++) {
            arr[2] = arr[0] + arr[1];
            arr[0] = arr[1];
            arr[1] = arr[2];
        }
        return arr[1];
    }
}

    //加强版斐波那契数
class Solution {
    public int tribonacci(int n) {
        if(n==0) return 0;
        if(n==1) return 1;
        if(n==2) return 1;
        int[] arr = new int[4];
        arr[0] = 0;
        arr[1] = 1;
        arr[2] = 1;
        for(int i=3;i<=n;i++) {
            arr[3] = arr[0] + arr[1] + arr[2];
            arr[0] = arr[1];
            arr[1] = arr[2];
            arr[2] = arr[3];
        }
        return arr[2];
    }
}

    # 1646. 获取生成数组中的最大值

    //这道题或许不能用滚动数组，需要将所有生成数保存起来
class Solution {
    public int getMaximumGenerated(int n) {
        if(n==0) return 0;
        if(n==1) return 1;
        int max_value = Integer.MIN_VALUE;
        int[] arr = new int[n+1];
        arr[0] = 0;
        arr[1] = 1;
        for(int i=2;i<=n;i++) {  //从2开始去dp计算
            if(i%2==0) { //偶数
                arr[i] = arr[i/2];
            } else { //奇数
                arr[i] = arr[i/2] + arr[i/2+1];
            }
            if(max_value<arr[i]) max_value = arr[i];
        }
        return max_value;
    }
}

    # LCP 07. 传递信息

    //从这道题开始就接触了有向图的题目
//首先用一个二维List存储有向边
//适当保存n和k的信息
//dfs递归函数的参数是当前start和已走的步数step
// 这道题开始接触有向图的题目, List<List<Integer>> edges 保存有向边
// 方法1 DFS深度优先
class Solution {
    private int n, k;
    private int res;
    List<List<Integer>> edges = new ArrayList<>();
    public int numWays(int n, int[][] relation, int k) {
        this.n = n;
        this.k = k;
        for(int i=0;i<n;i++) {
            edges.add(new ArrayList<Integer>());
        }
        for(int[] tmp: relation) { //保存有向边的信息
            edges.get(tmp[0]).add(tmp[1]);
        }
        dfs(0, 0);
        return res;
    }
    public void dfs(int start, int steps) { //当前位置 已走步数
        //如果当前已经达到步数
        if(steps>=this.k) {
            if(start==n-1) this.res++; //如果当前已经到达了n-1的位置说明ok
            return;
        } else {
            //还没到目标位置 dfs递归调用
            for(int i=0; i<edges.get(start).size(); i++) {
                dfs(edges.get(start).get(i), steps+1);
            }
        }
    }
}

    方法2 BFS 广度优先方法：将0节点加入到队列，然后让循环执行steps次，每次拿出队列的全部元素，并将这些元素可达的目标位置全部入队。然后steps次循环执行完成之后，将队列元素全部出队，有多少个n-1，就有多少条路径

    // BFS方法
class Solution {
    public int numWays(int n, int[][] relation, int k) {
        List<List<Integer>> edges = new ArrayList<>();
        for(int i=0;i<n;i++) {
            edges.add(new ArrayList<Integer>());
        }
        for(int[] tmp: relation) {
            edges.get(tmp[0]).add(tmp[1]);
        } //存储有向边信息
        LinkedList<Integer> queue = new LinkedList<Integer>();
        queue.addFirst(0);
        int steps = 0;
        while(!queue.isEmpty() && steps<k) {
            steps++;
            int length = queue.size();
            for(int i=0;i<length;i++) {
                int tmp = queue.removeLast();
                for(int j=0;j<edges.get(tmp).size();j++) {
                    queue.addFirst(edges.get(tmp).get(j));
                }
            }
        }
        int res = 0;
        while(!queue.isEmpty()) {
            if(queue.removeLast()==n-1) res++;
        }
        return res;
    }
}

    # LCS 01. 下载插件

    这道题本来在想什么"好办法"，看了解析后明白，最简单的思路是优先提高带宽，直到流量为下载总数的一半，然后再用2分钟下载即可

    //先加速 加速到 n/flow<2 的时候，只需要两分钟的时间下载即可 比如n=5  flow = 3
class Solution {
    public int leastMinutes(int n) {
        int flow = 1; //初始化流量
        int time = 0;
        while(n>0) {
            if(n/flow>2) { //带宽不够 继续提高带宽
                time++;
                flow *= 2; //带宽加倍
            } else { //带宽已够 开始下载
                time++;
                n -= flow;
            }
        }
        return time;
    }
}

    # 面试题 08.01. 三步问题

    这道题也是跟斐波那契数一样的逻辑+滚动数组 小孩可以一次走1/2/3步 求爬n阶楼梯的方法总数 过不了的原因在于越界溢出问题，这时候需要用Long，不要用int

    //f(n) = f(n-1) + f(n-2) + f(n-3)
class Solution {
    public int waysToStep(int n) {
        if(n==1) return 1;
        if(n==2) return 2;
        if(n==3) return 4;
        long[] arr = new long[4];
        arr[0] = 1;
        arr[1] = 2;
        arr[2] = 4;
        for(int i=4;i<=n;i++) {
            arr[3] = (arr[0] + arr[1] + arr[2]) % 1000000007;
            arr[0] = arr[1];
            arr[1] = arr[2];
            arr[2] = arr[3];
        }
        return (int)arr[2];
    }
}

    # 面试题 16.17. 连续数列

    给你一个元素可正可负的数组，求出最大子序列和 f(i)代表以nums[i]结尾的最大子序列和(即为一定要包括 i 元素)，这个最大子序列和要么是f(i-1)+nums[i] 要么是舍弃前面的，只是nums[i] 看他俩谁大即可 最大子序列和的初始化值应该是nums[0] pre保存f(i-1) 如果循环从index=0开始，那么pre初始化为0

    //思路还是对的 f(i)代表以nums[i] 结尾 的和最大的子序列
//f(i) = max(f(i), f(i-1)+nums[i]);  //4 -1 还是3比-1大
//滚动数组
class Solution {
    public int maxSubArray(int[] nums) {
        if(nums.length==0) return 0;
        if(nums.length==1) return nums[0];
        int res = nums[0];//最大子序和初始化为nums[0]
        int pre = 0; //保存f(i-1)
        for(int i=0;i<nums.length;i++) {
            pre = Math.max(pre+nums[i], nums[i]); //即便从i=0开始，pre=0时pre+nums[0]和nums[0]相等
            if(res<pre) res = pre;
        }
        return res;
    }
}

    # 面试题 17.16. 按摩师

    给你一个数组代表N个预约的分钟数，按摩师不能接相邻的预约，求可以接的最大总分钟数 这道题是一道很好的dp题，思路很好，每个点的状态有两个，每个点的状态求解只与前一个点的两个状态有关 好题！！！但是我没思路！！ 需要复习！！

    //dp题我是真的跪了
//因为每个预约都有接和不接两种选择，所以对于每个元素应该都有两个临时结果
//dp[i][0]表示考虑 前i个 预约，不接第i个预约时最长的分钟数,dp[i][1] 同上，但接第i个预约
//"假设已经把前i-1个dp值已经计算好了"是dp题的基本思路
//dp[i][0] = max(dp[i-1][0], dp[i-1][1]) 第i个不接，那么第i-1个接或不接，选总分钟数大的即可
//dp[i][1] = dp[i-1][0]+nums[i] 因为第i个要接 只能从i-1不接的情况下转移过来
class Solution {
    public int massage(int[] nums) {
        if(nums.length==0) return 0;
        if(nums.length==1) return nums[0];
        int dp0 = 0; //初始化不接 即为0分钟
        int dp1 = nums[0]; //初始化接 即为nums[0]的分钟数
        for(int i=1;i<nums.length;i++) { //因为只考虑第一个元素的情况已经作为初始值了 所以从考虑前2个元素开始考虑
            int tdp0 = Math.max(dp0, dp1); //第i个不接，那么第i-1个接或不接，选总分钟数大的即可
            int tdp1 = dp0 + nums[i];  //第i个要接 只能从i-1不接的情况下转移过来
            dp0 = tdp0;
            dp1 = tdp1;
        }
        return Math.max(dp0, dp1);
    }
}
